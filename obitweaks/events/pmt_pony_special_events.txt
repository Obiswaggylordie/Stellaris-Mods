country_event = {
	id = pmtspecial.110
	hide_window = yes
	is_triggered_only = yes

	trigger = {
		is_ai = no
		pmt_pony_species_trigger = yes
		NOT = { has_country_flag = pony_cult_evt }
		is_regular_empire = yes
		any_spynetwork = {
			owner = { is_same_value = root }
			target = {
				is_country_type = primitive
				is_gestalt = no
				is_pre_industrial = no
				NOT = { pmt_pony_species_trigger = yes }
				OR = {
					has_country_flag = late_medieval_age
					has_country_flag = renaissance_age
					has_country_flag = steam_age
					has_country_flag = industrial_age
					has_country_flag = machine_age
				}
				NOR = {
					has_ethic = ethic_xenophile
					has_ethic = ethic_fanatic_xenophile
					has_ethic = ethic_spiritualist
					has_ethic = ethic_fanatic_spiritualist
					has_ethic = ethic_xenophobe
					has_ethic = ethic_fanatic_xenophobe
				}
				has_country_flag = preftl_espionage_operation_targeted_by@root
				NOR = {
					has_country_flag = ongoing_preftl_espionage_chain
					is_broken_shackles_pre_ftl_to_scope = { SCOPE = root }
				}
			}
			is_running_espionage_operation = no
		}
	}

	immediate = {
		random_country = {
			limit = {
				is_country_type = primitive
				is_gestalt = no
				is_pre_industrial = no
				NOT = { pmt_pony_species_trigger = yes }
				OR = {
					has_country_flag = late_medieval_age
					has_country_flag = renaissance_age
					has_country_flag = steam_age
					has_country_flag = industrial_age
					has_country_flag = machine_age
				}
				NOR = {
					has_ethic = ethic_xenophile
					has_ethic = ethic_fanatic_xenophile
					has_ethic = ethic_spiritualist
					has_ethic = ethic_fanatic_spiritualist
					has_ethic = ethic_xenophobe
					has_ethic = ethic_fanatic_xenophobe
				}
				has_country_flag = preftl_espionage_operation_targeted_by@root
				NOR = {
					has_country_flag = ongoing_preftl_espionage_chain
					is_broken_shackles_pre_ftl_to_scope = { SCOPE = root }
				}
			}
			set_country_flag = ongoing_preftl_espionage_chain
			save_event_target_as = rogue_agent_country
			capital_scope = {
				save_event_target_as = rogue_agent_planet
			}
		}
		set_country_flag = pony_cult_evt
		country_event = { id = pmtspecial.111 }
	}
}

		random_owned_leader = {
			limit = {
				if = {
					limit = {
						owner = {
							has_country_flag = accredit_alicorn_commander
						}
					}
					leader_class = commander
				}
				else_if = {
					limit = {
						owner = {
							has_country_flag = accredit_alicorn_scientist
						}
					}
					leader_class = scientist
				}
				else_if = {
					limit = {
						owner = {
							has_country_flag = accredit_alicorn_official
						}
					}
					leader_class = official
				}
				NOT = { has_trait = leader_trait_pmt_alicorn }
				NOT = { has_leader_flag = accredit_alicorn_leader }
				species = {
					pmt_pony_species_trigger = yes
				}
			}
			if = {
				limit = { exists = this }
				save_event_target_as = accredit_alicorn_leader_6
				set_leader_flag = accredit_alicorn_leader
			}
		}
	}

	after = {
		country_event = { id = pmtspecial.1110 }
	}
