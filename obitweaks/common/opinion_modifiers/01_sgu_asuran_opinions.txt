triggered_opinion_asuran_ancient = {
	trigger = {
		NOT = { is_country_type = fallen_empire }
		has_trait = trait_asuran
		FROM = {
			has_trait = trait_ancient
		}
	}

	opinion = {
		base = 100
	}
}
