namespace = celestial_energy_events

# Cosmoreverse Activation
country_event = { # Planetary Energy
	id = celestial_energy_events.1
	title = "celestial_energy_events.1.title"
	desc = "celestial_energy_events.1.desc"
	picture = GFX_energy_controller
	show_sound = event_default
	location = event_target:ce_target
	
	hide_window = no
	is_triggered_only = yes
	
    option = { # Fill Energy 
        name = "celestial_energy_events.1.a"
		
		allow = {
			has_technology = tech_zeia_energy_transfer_module
			has_country_resource = {
				type = sr_celestial_energy
				amount >= 100
			}
			event_target:ce_target = {
				NOT = { is_planet_class = pc_gaia }
			}
		}

        hidden_effect = {
			add_resource = {
				sr_celestial_energy = -100
			}
            event_target:ce_target = {
				celestial_energy_fill_planet_effect = yes
            }
        }
    }

    option = { # Drain Energy
        name = "celestial_energy_events.1.b"
		
		allow = {
			owner = {
				has_technology = tech_zeia_energy_transfer_module
			}
			event_target:ce_target = {
				NOR = {
					is_planet_class = pc_barren
					is_planet_class = pc_barren_cold
					is_planet_class = pc_frozen
					is_planet_class = pc_toxic
					is_planet_class = pc_molten
					is_planet_class = pc_shattered
					is_planet_class = pc_shattered_2
				}
			}
		}

        hidden_effect = {
			add_resource = {
				sr_celestial_energy = 100
			}
            event_target:ce_target = {
				celestial_energy_drain_planet_effect = yes
            }
        }
    }

	option = { # Light Shafts
		name = "celestial_energy_events.1.c"

		allow = {
			has_technology = tech_zeia_light_shaft
		}

		hidden_effect = {
			event_target:ce_target = {
				add_deposit = d_light_shaft
			}
		}
	}

	option = { # Cancel
		name = "celestial_energy_events.1.z"
	}
}
