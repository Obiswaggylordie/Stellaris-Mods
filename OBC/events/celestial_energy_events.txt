			add_resource = {
				sr_celestial_energy = -100
			}
            event_target:ce_target = {
				if = { # Warm
					limit = {
						has_planet_flag = ce_warm
					}
					if = {
						limit = {
							is_planet_class = pc_tropical
						}
						change_pc = pc_gaia

					else_if = {
						limit = {
							is_planet_class = pc_continental

					}
					else_if = {
						limit = {
							is_planet_class = pc_arctic
						}
						change_pc = pc_tundra
					}
					else_if = {
						limit = {
							is_planet_class = pc_barren_cold
						}
						change_pc = pc_arctic
					}
				}
				else_if = { # Cold
					limit = {
						has_planet_flag = ce_cold
					}
					if = {
						limit = {
							is_planet_class = pc_tropical
						}
						change_pc = pc_gaia
					}
					else_if = {
						limit = {
							is_planet_class = pc_ocean

					}
					else_if = {
						limit = {
							is_planet_class = pc_frozen
						}
						change_pc = pc_arctic
					}
				}
				
            }
        }
    }

    option = { # Drain Energy
        name = "celestial_energy_events.1.b"

			owner = {

			}

					else_if = {

						}

					}
				}
				else_if = { # Average
					limit = {
						has_planet_flag = ce_average
					}
					if = {
						limit = {
							is_planet_class = pc_gaia
						}
						change_pc = pc_tropical
					}
					else_if = {
						limit = {
							is_planet_class = pc_tropical
						}

					}
					else_if = {
						limit = {
							is_planet_class = pc_continental
						}

					}
					else_if = {

					}
					else_if = {
						limit = {
							is_planet_class = pc_tundra
						}
						change_pc = pc_arctic
					}
					else_if = {
						limit = {
							is_planet_class = pc_arctic
						}
						change_pc = pc_barren_cold
						destroy_colony = yes
					}
					
				}
				else_if = { # Cold
					limit = {

					}

		name = "celestial_energy_events.1.c"

		allow = {
			has_technology = tech_zeia_light_shaft
		}

	}
