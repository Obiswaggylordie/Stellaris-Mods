# Relay Management Menu
country_event = {
	id = void_relay.1
	title = "void_relay.1.title"
	desc = "void_relay.1.desc"
	show_sound = event_default
	
	hide_window = no
	is_triggered_only = yes
	
	option = { # Deactivate all Relays
		name = "void_relay.1.a"
		hidden_effect = {
			every_megastructure = {
				limit = {
					OR = {
						is_megastructure_type = void_relay
						is_megastructure_type = void_relay_alpha
						is_megastructure_type = void_relay_beta
						is_megastructure_type = void_relay_delta
						is_megastructure_type = void_relay_gamma
						is_megastructure_type = void_relay_lambda
						is_megastructure_type = void_relay_epsilon
					}
				}
				if = {
					limit = {
						is_megastructure_type = void_relay
					}
					upgrade_megastructure_to = void_relay_omega_inactive
				}
				else_if = {
					limit = {
						is_megastructure_type = void_relay_alpha
					}
					upgrade_megastructure_to = void_relay_alpha_inactive
				}
				else_if = {
					limit = {
						is_megastructure_type = void_relay_beta
					}
					upgrade_megastructure_to = void_relay_beta_inactive
				}
				else_if = {
					limit = {
						is_megastructure_type = void_relay_delta
					}
					upgrade_megastructure_to = void_relay_delta_inactive
				}
				else_if = {
					limit = {
						is_megastructure_type = void_relay_gamma
					}
					upgrade_megastructure_to = void_relay_gamma_inactive
				}
				else_if = {
					limit = {
						is_megastructure_type = void_relay_lambda
					}
					upgrade_megastructure_to = void_relay_lambda_inactive
				}
				else_if = {
					limit = {
						is_megastructure_type = void_relay_epsilon
					}
					upgrade_megastructure_to = void_relay_epsilon_inactive
				}
			}
		}
	}

	option = { # Activate all Relays
		name = "void_relay.1.b"

		hidden_effect = {
			every_megastructure = {
				limit = {
					OR = {
						is_megastructure_type = void_relay_omega_inactive
						is_megastructure_type = void_relay_alpha_inactive
						is_megastructure_type = void_relay_beta_inactive
						is_megastructure_type = void_relay_delta_inactive
						is_megastructure_type = void_relay_gamma_inactive
						is_megastructure_type = void_relay_lambda_inactive
						is_megastructure_type = void_relay_epsilon_inactive
					}
				}
				if = {
					limit = {
						is_megastructure_type = void_relay_omega_inactive
					}
					upgrade_megastructure_to = void_relay
				}
				else_if = {
					limit = {
						is_megastructure_type = void_relay_alpha_inactive
					}
					upgrade_megastructure_to = void_relay_alpha
				}
				else_if = {
					limit = {
						is_megastructure_type = void_relay_beta_inactive
					}
					upgrade_megastructure_to = void_relay_beta
				}
				else_if = {
					limit = {
						is_megastructure_type = void_relay_delta_inactive
					}
					upgrade_megastructure_to = void_relay_delta
				}
				else_if = {
					limit = {
						is_megastructure_type = void_relay_gamma_inactive
					}
					upgrade_megastructure_to = void_relay_gamma
				}
				else_if = {
					limit = {
						is_megastructure_type = void_relay_lambda_inactive
					}
					upgrade_megastructure_to = void_relay_lambda
				}
				else_if = {
					limit = {
						is_megastructure_type = void_relay_epsilon_inactive
					}
					upgrade_megastructure_to = void_relay_epsilon
				}
			}
		}
	}

	option = { # Access Relay Management
		name = "void_relay.1.c"
		hidden_effect = {
			country_event = { id = void_relay.2 }
		}
	}

	option = { # Assume control of all Relays
		name = "void_relay.1.d"
		hidden_effect = {
			this = {
				save_global_event_target_as = relay_controller
			}
			every_megastructure = {
				limit = {
					OR = {
						is_megastructure_type = void_relay
						is_megastructure_type = void_relay_omega_inactive
						is_megastructure_type = void_relay_alpha
						is_megastructure_type = void_relay_alpha_inactive
						is_megastructure_type = void_relay_beta
						is_megastructure_type = void_relay_beta_inactive
						is_megastructure_type = void_relay_delta
						is_megastructure_type = void_relay_delta_inactive
						is_megastructure_type = void_relay_gamma
						is_megastructure_type = void_relay_gamma_inactive
						is_megastructure_type = void_relay_lambda
						is_megastructure_type = void_relay_lambda_inactive
						is_megastructure_type = void_relay_epsilon
						is_megastructure_type = void_relay_epsilon_inactive	
						is_megastructure_type = void_relay_deactivate		
					}
				}
				set_owner = event_target:relay_controller
			}
		}
	}

	option = { # Access Perdition & Genesis
		name = "void_relay.1.x"

		allow = {
			any_owned_megastructure = {
				is_megastructure_type = crucible
			}
		}

		hidden_effect = {
			country_event = { id = perdition_genesis.1 }
		}
	}
	
	option = { # Disable Relay Control
		name = "void_relay.1.y"

		allow = {
			event_target:obicitadel = {
				has_deposit = d_citadel_relay_control
			} 
		}
		
		hidden_effect = {
			event_target:obicitadel = {
				remove_deposit = d_citadel_relay_control
				add_deposit = d_citadel_inactive_relay_control
			}
			
			add_resource = {
				sr_sdc = 1
			}
		}
	}

	option = { # Exit Menu
		name = "void_relay.1.z"
	}
}

	option = { # Omega to Alpha
		name = "void_relay.4.a"
		hidden_effect = {

				limit = {
					AND = {
						is_megastructure_type = void_relay_deactivate
