#ENERGON SPECIALISTS
cybertronian_energon_refiner = {	
    category = specialist
	swappable_data = {
		default = {
			condition_string = SPECIALIST_JOB_TRIGGER
			building_icon = building_cybertronian_energon_refinery
		}
	}
    #clothes_texture_index = 3
    tags = { strategic_resource energon }
	
    possible_pre_triggers = {
		has_owner = yes
		is_being_purged = no
		is_being_assimilated = no
		is_sapient = yes
	}

	possible_precalc = can_fill_specialist_job
	
	resources = {
		category = planet_jobs
		produces = {
			energon = 12
			#energy = 10
		}
		upkeep = {
			energon_crystals = 10
        }
    }

    weight = {
		weight = @specialist_job_weight
		modifier = {
			factor = 2.0
			OR = {
				has_trait = trait_cybertronian
				has_trait = trait_bw_cybertronian
				has_trait = trait_transtech_cybertronian
				has_trait = trait_energon_dependence
			}
		}
	}
	promotion = {
		time = @standard_promotion_time
		effect = {
			pop_amount_change_category = {
				CATEGORY = specialist
				AMOUNT = local_pop_amount
			}
		}
	}
}

#ENERGON WORKERS

energon_miner = {
	category = worker
	swappable_data = {
		default = {
			condition_string = WORKER_JOB_TRIGGER
			building_icon = building_energon_mine
		}
	}
	#clothes_texture_index = 2
	tags = { strategic_resource energon }
	
	possible_pre_triggers = {
		has_owner = yes
		is_being_purged = no
		is_being_assimilated = no
		is_sapient = yes
	}
	
	possible_precalc = can_fill_worker_job
	possible = {
		hidden_trigger = { exists = owner }
		owner = { is_gestalt = no }
	}
	
	resources = {
		category = planet_sr_miners
		produces = {
			energon_crystals = 10
		}
		upkeep = {
			energy = 1
		}
	}
	weight = {
		weight = @specialist_job_weight # Important job
		modifier = {
			factor = 5.0
			OR = {
				has_trait = trait_cybertronian
				has_trait = trait_bw_cybertronian
				has_trait = trait_transtech_cybertronian
				has_trait = trait_energon_dependence
			}
		}
		# low income modifier
		modifier = {
			factor = 3
			exists = planet
			planet = {
				has_available_jobs = energon_miner
			}
			exists = owner
			owner = {
				has_monthly_income = {
					resource = energon_crystals
					value < 10
				}
			}
		}
	}
	promotion = {
		time = @standard_promotion_time
		effect = {
			pop_amount_change_category = {
				CATEGORY = worker
				AMOUNT = local_pop_amount
			}
		}
	}
}
