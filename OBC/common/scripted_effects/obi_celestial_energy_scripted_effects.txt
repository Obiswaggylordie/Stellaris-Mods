# Warm: 	Gaia - Tropical - Continental - Arid - 	Savanna - Desert - Barren (Dry)
# Average: 	Gaia - Tropical - Continental - Alpine - Tundra - Arctic - Barren (Cold)
# Cold: 	Gaia - Tropical - Ocean - 		Alpine - Tundra - Arctic - Frozen 

# Class O - B - A - F - G - K - M - M Giant - T
# Dropping O for now.

celestial_energy_type_effect = {
	planet = {
		if = {
			limit = {
				is_planet_class = pc_gaia
			}
			random_list = {
				33 = { set_planet_flag = ce_warm }
				33 = { set_planet_flag = ce_average }
				33 = { set_planet_flag = ce_cold }
			}
		}
		else_if = {
			limit = {
				is_planet_class = pc_tropical
			}
			random_list = {
				33 = { set_planet_flag = ce_warm }
				33 = { set_planet_flag = ce_average }
				33 = { set_planet_flag = ce_cold }
			}
		}
		else_if = {
			limit = {
				is_planet_class = pc_continental
			}
			random_list = {
				50 = { set_planet_flag = ce_warm }
				50 = { set_planet_flag = ce_average }
			}
		}
		else_if = {
			limit = {
				is_planet_class = pc_ocean
			}
			set_planet_flag = ce_cold
		}
		else_if = {
			limit = {
				is_planet_class = pc_arid
			}
			set_planet_flag = ce_warm
		}
		else_if = {
			limit = {
				is_planet_class = pc_alpine
			}
			random_list = {
				50 = { set_planet_flag = ce_average }
				50 = { set_planet_flag = ce_cold }
			}
		}
		else_if = {
			limit = {
				is_planet_class = pc_savannah
			}
			set_planet_flag = ce_warm
		}
		else_if = {
			limit = {
				is_planet_class = pc_tundra
			}
			random_list = {
				50 = { set_planet_flag = ce_average }
				50 = { set_planet_flag = ce_cold }
			}
		}
		else_if = {
			limit = {
				is_planet_class = pc_desert
			}
			set_planet_flag = ce_warm
		}
		else_if = {
			limit = {
				is_planet_class = pc_arctic
			}
			random_list = {
				50 = { set_planet_flag = ce_average }
				50 = { set_planet_flag = ce_cold }
			}
		}
		else_if = {
			limit = {
				is_planet_class = pc_barren
			}
			set_planet_flag = ce_warm
		}
		else_if = {
			limit = {
				is_planet_class = pc_barren_cold
			}
			set_planet_flag = ce_average
		}
		else_if = {
			limit = {
				is_planet_class = pc_frozen
			}
			set_planet_flag = ce_cold
		}
	}
}

celestial_energy_fill_planet_effect = {
	planet = {
		if = { # Warm
			limit = {
				has_planet_flag = ce_warm
			}
			if = {
				limit = {
					is_planet_class = pc_tropical
				}
				change_pc = pc_gaia
			}
			else_if = {
				limit = {
					is_planet_class = pc_continental
				}
				change_pc = pc_tropical
			}
			else_if = {
				limit = {
					is_planet_class = pc_arid
				}
				change_pc = pc_continental
			}
			else_if = {
				limit = {
					is_planet_class = pc_savannah
				}

			}
			else_if = {
				limit = {
					is_planet_class = pc_desert
				}

			}
			else_if = {
				limit = {
					is_planet_class = pc_barren
				}
				change_pc = pc_desert
			}
		}
		else_if = { # Average
			limit = {
				has_planet_flag = ce_average
			}
			if = {
				limit = {
					is_planet_class = pc_tropical
				}
				change_pc = pc_gaia
			}
			else_if = {
				limit = {
					is_planet_class = pc_continental
				}
				change_pc = pc_tropical
			}
			else_if = {
				limit = {

				}

			}
			else_if = {

			}
			else_if = {
				limit = {
					is_planet_class = pc_barren_cold
				}
				change_pc = pc_arctic
			}
		}
		else_if = { # Cold
			limit = {
				has_planet_flag = ce_cold
			}
			if = {
				limit = {
					is_planet_class = pc_tropical
				}
				change_pc = pc_gaia
			}
			else_if = {
				limit = {
					is_planet_class = pc_ocean
				}
				change_pc = pc_tropical
			}
			else_if = {
				limit = {
					is_planet_class = pc_alpine
				}
				change_pc = pc_ocean
			}
			else_if = {
				limit = {
					is_planet_class = pc_tundra
				}
				change_pc = pc_alpine
			}
			else_if = {
				limit = {
					is_planet_class = pc_arctic
				}
				change_pc = pc_tundra
			}
			else_if = {
				limit = {
					is_planet_class = pc_frozen
				}
				change_pc = pc_arctic
			}
		}
		else_if = {
			limit = {
				NOT = {
					has_planet_flag = ce_warm
					has_planet_flag = ce_average
					has_planet_flag = ce_cold
				}
			}
			celestial_energy_type_effect = yes
			celestial_energy_fill_planet_effect = yes
		}
	}
}
